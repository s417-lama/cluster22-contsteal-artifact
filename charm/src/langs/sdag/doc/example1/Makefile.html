INC=-I../../runtime
LIB=-L../../runtime

all: cifiles sdagfiles pgm

cifiles:
        charmxi main.ci
        charmxi test.ci

sdagfiles:
        sdagx test.sdag

pgm: main.o test.o
        charmc -g -language charm++ -o pgm main.o test.o $(LIB) -lsdag

main.o: main.C main.h
        charmc -g -c main.C $(INC)

test.o: test.C test.h
        charmc -g -c test.C $(INC)

clean:
        rm -f pgm *.o test.sdag.h test.sdag.C *.top.h *.bot.h
